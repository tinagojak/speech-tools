38  /* associated with enums, providing conversion. */

80 template<class ENUM, class VAL, class INFO> class EST_TValuedEnumI {
81 
82 protected:

85  VAL p_unknown_value;

87

89  // This is only a void * because INFO can`t manage to get the
90  // parameter declaration in the definition past gcc with the actual type.
91  void initialise(const void *defs);
92  void initialise(const void *defs, ENUM (*conv)(const char *));
93  void initialise(void) {ndefinitions=0; definitions=NULL;};

96 protected:

105
