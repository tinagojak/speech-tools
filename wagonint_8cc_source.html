49 int wagon_ask_question(LISP question, LISP value)
50 {
51  const char *str_oper = wgn_ques_oper_str(question);
52 
53  if (streq("is",str_oper))
54  {
55  if (streq(get_c_string(value),
56  get_c_string(wgn_ques_operand(question))))
57  return TRUE;
58  else
59  return FALSE;
60  }
61  else if (streq("=",str_oper))
62  if (get_c_float(value) == get_c_float(wgn_ques_operand(question)))
63  return TRUE;
64  else
65  return FALSE;
66  else if (streq("<",str_oper))
67  if (get_c_float(value) < get_c_float(wgn_ques_operand(question)))
68  return TRUE;
69  else
70  return FALSE;
71  else if (streq(">",str_oper))
72  if (get_c_float(value) > get_c_float(wgn_ques_operand(question)))
73  return TRUE;
74  else
75  return FALSE;
76  else if (streq("matches",str_oper))
77  {
78  EST_Regex rgx(get_c_string(wgn_ques_operand(question)));
79  if (EST_String(get_c_string(value)).matches(rgx))
80  return TRUE;
81  else
82  return FALSE;
83  }
84  else if (streq("in",str_oper))
85  if (siod_member_str(get_c_string(value),
86  wgn_ques_operand(question)) != NIL)
87  return TRUE;
88  else
89  return FALSE;
90  else
91  {
92  cerr << "WAGON: unknown question operator: \"" <<
93  str_oper << "\"\n";
94  siod_error();
95  }
96  // not reached, just for the compiler
97  return FALSE;
98 }
