55 void EST_pre_emphasis(EST_Wave &signal, EST_Wave &psignal, float a)
56 {
57  int num_samples = signal.num_samples();
58  short last=0;
59  double maxval=0;
60  EST_TBuffer<double> fpdata(num_samples);
61 
62  for(int i=0; i<num_samples; i++)
63  {
64  fpdata[i] = signal.a(i) - a*last;
65  last = signal.a(i);
66  if (absval(fpdata[i]) > maxval)
67  maxval = absval(fpdata[i]);
68  }
69 
70  psignal.resize(num_samples,1,FALSE);
71  psignal.set_sample_rate(signal.sample_rate());
72  short_set(psignal, 0, fpdata, maxval);
73 }

76 void EST_post_deemphasis(EST_Wave &signal, EST_Wave &dsignal, float a)
77 {
78  int num_samples = signal.num_samples();
79  double last=0;
80  double maxval=0;
81  EST_TBuffer<double> fddata(num_samples);
82 
83  for(int i=0; i<num_samples; i++)
84  {
85  fddata[i] = signal.a(i) + a*last;
86  last = fddata[i];
87  if (absval(fddata[i]) > maxval)
88  maxval = absval(fddata[i]);
89  }
90 
91  dsignal.resize(num_samples,1,FALSE);
92  dsignal.set_sample_rate(signal.sample_rate());
93  short_set(dsignal, 0, fddata, maxval);
94 }
