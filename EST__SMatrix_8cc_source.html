64 int EST_SMatrix::rateconv(int in_samp_freq, int out_samp_freq)
65 {
66  short *in_buf = new short[num_rows()];
67  short ** results = new short *[num_columns()];
68  int *len = new int[num_columns()];
69  int max_len=0;
70 
71  for(ssize_t c=0; c<num_columns(); c++)
72  {
73  short *out_buf;
74  int osize;
75 
76  copy_column(c, in_buf);
77 
78  if (::rateconv(in_buf,
79  num_rows(), &out_buf, &osize,
80  in_samp_freq, out_samp_freq) == 0)
81  {
82  results[c]=out_buf;
83  len[c]=osize;
84  if (osize > max_len)
85  max_len = osize;
86  }
87  else {
88  delete [] results;
89  delete [] len;
90  return -1;
91  }
92  }
93  delete [] in_buf;
94 
95  resize(max_len, EST_CURRENT, 0);
96  fill(0);
97 
98  for(ssize_t c1=0; c1<num_columns(); c1++)
99  {
100  set_column(c1, results[c1], 0, len[c1]);
101  delete [] results[c1];
102  }
103 
104  delete [] results;
105  delete [] len;
106  return 0;
107 }
