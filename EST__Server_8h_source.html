41 /** \class EST_Server
42  * \brief Client-server interface.
43  *
44  * An EST_Server object represents a server, it can be at either
45  * end of a network connection. That is, a server process has an
46  * EST_Server object representing it's wait-dispatch-answer loop,
47  * while a client process has an EST_Server object which represents
48  * the server process.
49  *
50  * @author Richard Caley <rjc@cstr.ed.ac.uk>
51  * @version $Id: EST_Server.h,v 1.4 2004/05/04 00:00:16 awb Exp $
52  */
53 class EST_Server {
54

57  /// What type of server is this.
58  enum Mode {
59  /// Bizarre state
60  sm_unknown = 0,
61  /// Client end of the connection.
62  sm_client = 1,
63  /// Answer one client at a time.
64  sm_sequential = 2,
65  /// Answer requests from several clients, as requests arrive.
66  sm_interleaved =3,
67  /// For off a process for each client.
68  sm_fork = 4,
69  /// Multi-threaded (not implemented)
70  sm_threded = 5
71  };
