56 void value_sort(EST_Features &f, const EST_String &field)
57 {
58  int work_to_do = 1;
59  (void)field;
60 
61  EST_Features::RwEntries p;
62  EST_Features::RwEntries n;
63  EST_Features::Entry e;
64 
65  float p_score, n_score;
66 
67  while(work_to_do)
68  {
69  work_to_do = 0;
70  for (p.begin(f); p; ++p)
71  {
72  n = p;
73  n++;
74  if (n == 0)
75  break;
76 
77  if (p->v.type() != val_type_feats)
78  {
79  cerr << "Not a features in val\n";
80  break;
81  }
82  p_score = feats(p->v)->F(field, 1.0);
83  n_score = feats(n->v)->F(field, 1.0);
84  if (n_score > p_score)
85  {
86  cout << "swapping\n";
87  e = *p;
88  *p = *n;
89  *n = e;
90  work_to_do = 1;
91  }
92  }
93  }
94 }
