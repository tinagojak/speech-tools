37 /* XXX Tolower and Toupper only work for ISO-Latin-1 */
38 
39 int Tolower(int c)
40 {
41  if(c < 0) /* Sigh, char is probably signed */
42  c &= 0xff;
43 
44  if((c >= 'A' && c <= 'Z') ||
45  (c >= 192 && c <= 214) ||
46  (c >= 216 && c <= 222))
47  return c + 32;
48  else
49  return c;
50 }
51 
52 int Toupper(int c)
53 {
54  if(c < 0) /* Sigh, char is probably signed */
55  c &= 0xff;
56 
57  if((c >= 'a' && c <= 'z') ||
58  (c >= 224 && c <= 246) ||
59  (c >= 248 && c <= 254))
60  return c - 32;
61  else
62  return c;
63 }
