59 EST_FMatrix confusion(EST_StrStr_KVL &list, EST_StrList &lex)
60 {
61  EST_FMatrix a(lex.length(), lex.length());
62  EST_Litem *p;
63  int n, m;
64  a.fill(0.0);
65 
66  for (p = list.list.head(); p!=0; p = p->next())
67  {
68  m = nth(list.key(p), lex);
69  n = nth(list.val(p), lex);
70  if ((n != -1) && (m != -1))
71  a(m, n) = a(m, n) + 1;
72  }
73 
74  return a;
75 }
