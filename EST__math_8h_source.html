158 inline double safe_log(const double x)
159 {
160  double l;
161  if (x == 0)
162  return SAFE_LOG_ZERO;
163  l=log(x);
164  if (l<SAFE_LOG_ZERO)
165  return SAFE_LOG_ZERO;
166  else
167  return l;
168 }

170 inline double safe_exp(const double x)
