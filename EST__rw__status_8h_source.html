62 #define make_status_int(STATE, REASON, ERRNO) \
63  ( ((((unsigned int)(STATE)) & 0xff) <<24) \
64  | ((((unsigned int)(REASON)) & 0xff) <<16) \
65  | ((((unsigned int)(ERRNO)) & 0xffff) <<00) \
66  )
