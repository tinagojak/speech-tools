40 /** @class EST_Handleable
41  * @ingroup supportclasses
42  * @brief Reference Counting Interface.
43  *
44  * This simple class does most of the things an object which is to be
45  * manipulated by EST_THandle style smart pointers needs to provide.
46  *
47  * @see EST_THandle
48  * @see EST_TBox
49  * @see EST_TrackMap
50  * @author Richard Caley <rjc@cstr.ed.ac.uk>
51  * @version $Id: EST_Handleable.h,v 1.4 2004/09/29 08:24:17 robert Exp $
52  */
53 class EST_Handleable
54 {
55 private:
56  int p_refcount;
57 
58 public:
59 # define NOT_REFCOUNTED (INT_MAX)
60 
61  EST_Handleable(void) { p_refcount=NOT_REFCOUNTED; }
62 
63  int refcount(void) const { return p_refcount;}
64 
65  void start_refcounting(int initial=0) {p_refcount=initial;}
66  void inc_refcount(void) {if (p_refcount!=NOT_REFCOUNTED) p_refcount++;}
67  void dec_refcount(void) {if (p_refcount!=NOT_REFCOUNTED) p_refcount--;}
68 
69  int is_unreferenced(void) const {return p_refcount == 0;}
70  int is_refcounted(void) const {return p_refcount!= NOT_REFCOUNTED;}
71  //@}
72 };
